\def\modele@uranus{uranus}
\def\cv@h@uranus{5}

% =============== FOND ===============
\newcommand*\fond@uranus{%
	\ifx\cmdPAS@bg@pic\@empty%
	\else
		\ifx\cmdPAS@bg@borderpic\@empty
			\node[below] at (3.25,-.5) {\includegraphics[scale=\cmdPAS@bg@scalepic]{\cmdPAS@bg@pic}};
		\else
			\node[below,draw=\cmdPAS@bg@borderpic,very thick,inner sep=1pt] at (3.25,-.5) {\includegraphics[scale=\cmdPAS@bg@scalepic]{\cmdPAS@bg@pic}};
		\fi
	\fi
	\node[color=\cmdPAS@bg@postecolor,left] at (.98\paperwidth,-\cv@h@uranus+.25) {\scshape\cmdPAS@bg@postesize\bfseries\cmdPAS@bg@poste};
	\clip (0,0) -- (2,0) .. controls (1.5,-1) and (1.5,-2) .. (2,-5) .. controls (5,-15) and (5,-20) .. (3,-29.7) -- (0,-29.7) -- cycle;
	% 1èle portion
	\fill[\cmdPAS@bg@bgcolor] (0,0) rectangle+(5,-6);
	\shade[right color=\cmdPAS@bg@bgcolor,left color=\cmdPAS@bg@bordercolor] (0,0) rectangle+(2.5,-6);
	% 2ème portion
	\fill[fill=\cmdPAS@bg@bgcolor!50!black] (0,-7) rectangle+(5,-8);
	\shade[right color=\cmdPAS@bg@bgcolor!50!black,left color=\cmdPAS@bg@bordercolor] (0,-7) rectangle+(3.5,-8);
	% 3ème portion
	\fill[fill=\cmdPAS@bg@bgcolor!25!white] (0,-16) rectangle+(5,-6);
	\shade[right color=\cmdPAS@bg@bgcolor!25!white,left color=\cmdPAS@bg@bordercolor] (0,-16) rectangle+(3,-6);
	% 4ème portion
	\fill[fill=\cmdPAS@bg@bgcolor!75!black] (0,-23) rectangle+(5,-7);
	\shade[right color=\cmdPAS@bg@bgcolor!75!black,left color=\cmdPAS@bg@bordercolor] (0,-23) rectangle+(3,-7);
	\clip (0,0) -- (.7,0) .. controls (2,-10) and (3,-17) .. (1,-29.7) -- (0,-29.7) -- cycle;
	\fill[white] (0,0) rectangle+(5,-29.7);
}

%=============== INFO GAUCHE ===============
\newcommand*\infoLeft@uranus[1]{%
	\AddToShipoutPicture{%
		\put(\LenToUnit{5cm},\LenToUnit{.97\paperheight})
			{%
				\rlap{
					\begin{minipage}[t]{0.5\paperwidth}
						#1
					\end{minipage}
				}
			}
	}
}

%=============== INFO DROITE ===============
\newcommand*\infoRight@uranus[1]{%
	\AddToShipoutPicture{%
		\put(\LenToUnit{.97\paperwidth},\LenToUnit{.97\paperheight}){%
			\llap{%
			\begin{minipage}[t]{.5\paperwidth}
				\begin{flushright}#1\end{flushright}
			\end{minipage}
			}
		}
	}
}

%=============== TITRE ===============
\newcommand*\title@uranus[1]{%
	\noindent
	\begin{minipage}{1cm}
	\begin{tikzpicture}
		\node[left,text=\cmdPAS@title@color,rotate=90] (title) {\scshape\bfseries #1};
		\draw[\cmdPAS@title@bordercolor,very thick] (title.south east) -- (title.south west);
	\end{tikzpicture}
	\end{minipage}
}
