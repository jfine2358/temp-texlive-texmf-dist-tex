\starttypescript [sans] [uqammaq]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Uqammaq-Regular] [features=default]
    \definefontsynonym [SansBold] [Uqammaq-Bold]    [features=default]
\stoptypescript

\starttypescript [sans] [uqammaq-light]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Uqammaq-Light]   [features=default]
    \definefontsynonym [SansBold] [Uqammaq-Regular] [features=default]
\stoptypescript

\starttypescript [sans] [uqammaq-bold]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Uqammaq-Bold]  [features=default]
    \definefontsynonym [SansBold] [Uqammaq-Heavy] [features=default]
\stoptypescript

\starttypescript [sans] [uqammaq,uqammaq-light,uqammaq-bold]
    \definefontsynonym [Uqammaq-Light]   [name:Uqammaq Light] [features=default]
    \definefontsynonym [Uqammaq-Regular] [name:Uqammaq]       [features=default]
    \definefontsynonym [Uqammaq-Bold]    [name:Uqammaq Bold]  [features=default]
    \definefontsynonym [Uqammaq-Heavy]   [name:Uqammaq Heavy] [features=default]
\stoptypescript

\starttypescript [uqammaq,uqammaq-light,uqammaq-bold]
    \definetypeface [\typescriptone] [ss] [sans] [\typescriptone] [default]
    \quittypescriptscanning
\stoptypescript

\endinput
