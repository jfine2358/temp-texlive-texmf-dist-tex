\starttypescriptcollection [smeltery]

% AUdimat

\starttypescript [sans] [audimat]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [Audimat-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [Audimat-Italic]     [features=default]
    \definefontsynonym [SansBold]       [Audimat-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [Audimat-BoldItalic] [features=default]
\stoptypescript

\starttypescript [mono] [audimat]
    \setups[font:fallback:mono]
    \definefontsynonym [Mono]           [AudimatMono-Regular]    [features=default]
    \definefontsynonym [MonoItalic]     [AudimatMono-Italic]     [features=default]
    \definefontsynonym [MonoBold]       [AudimatMono-Bold]       [features=default]
    \definefontsynonym [MonoBoldItalic] [AudimatMono-BoldItalic] [features=default]
\stoptypescript

\starttypescript [audimat]
    \definetypeface [audimat] [ss] [sans] [audimat] [default]
    \definetypeface [audimat] [tt] [mono] [audimat] [default]
    \quittypescriptscanning
\stoptypescript

% GEronto Bis

\starttypescript [serif] [geronto]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif] [Geronto-Regular] [features=default]
\stoptypescript

\starttypescript [geronto]
    \definetypeface [geronto] [rm] [serif] [geronto] [default]

\stoptypescript
