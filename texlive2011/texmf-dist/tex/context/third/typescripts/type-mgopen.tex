\starttypescriptcollection [mgopen]

% Canonica

\starttypescript [serif] [mgopen-canonica]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [MgOpenCanonica-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [MgOpenCanonica-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [MgOpenCanonica-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [MgOpenCanonica-BoldItalic] [features=default]
\stoptypescript

\starttypescript [mgopen-canonica]
    \definetypeface [mgopen-canonica] [rm] [serif] [mgopen-canonica] [default]
    \quittypescriptscanning
\stoptypescript
