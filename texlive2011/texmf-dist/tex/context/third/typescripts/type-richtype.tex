\starttypescriptcollection [richtype]

% Francesca

\starttypescript [serif] [francesca]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]       [Francesca-Regular] [features=default]
    \definefontsynonym [SerifBold]   [Francesca-Bold]    [features=default]
    \definefontsynonym [SerifItalic] [Francesca-Italic]  [features=default]
    \definefontsynonym [SerifCaps]   [Serif]             [features=smallcaps]
\stoptypescript

\starttypescript [francesca]
    \definetypeface [francesca] [rm] [serif] [francesca] [default]
    \quittypescriptscanning
\stoptypescript

% Klara

\starttypescript [serif] [klara]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]       [KlaraSerif-Regular]
    \definefontsynonym [SerifBold]   [KlaraSerif-Bold]
    \definefontsynonym [SerifItalic] [KlaraSerif-Italic]
\stoptypescript

\starttypescript [sans] [klara]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]       [KlaraSans-Regular]
    \definefontsynonym [SansBold]   [KlaraSans-Bold]
    \definefontsynonym [SansItalic] [KlaraSans-Italic]
\stoptypescript

\starttypescript [klara]
    \definetypeface [klara] [rm] [serif] [klara] [default]
    \definetypeface [klara] [ss] [sans]  [klara] [default]
    \quittypescriptscanning
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-richtype}
