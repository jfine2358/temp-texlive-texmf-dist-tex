\starttypescriptcollection [gfs]

% Ambrosia

\starttypescript [serif] [ambrosia]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif] [Ambrosia-Regular] [features=default]
\stoptypescript

\starttypescript [ambrosia]
    \definetypeface [ambrosia] [rm] [serif] [ambrosia] [default]
    \quittypescriptscanning
\stoptypescript
