\starttypescriptcollection [fontsite]

% Bergamo

\starttypescript [serif] [bergamo]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [Bergamo-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [Bergamo-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [Bergamo-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [Bergamo-BoldItalic] [features=default]
\stoptypescript

\starttypescript [bergamo]
    \definetypeface [bergamo] [rm] [serif] [bergamo] [default]
    \quittypescriptscanning
\stoptypescript

% Carto Gothic

\starttypescript [sans] [cartogothic]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [CartoGothic-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [CartoGothic-Italic]     [features=default]
    \definefontsynonym [SansBold]       [CartoGothic-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [CartoGothic-BoldItalic] [features=default]
\stoptypescript

\starttypescript [cartogothic]
    \definetypeface [cartogothic] [ss] [sans] [cartogothic] [default]
    \quittypescriptscanning
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-fontsite}
