\def\ArmTeX@Armtoday{\number\day~\ifcase\month\or
  hunvari\or phetrvari\or marti\or aprili\or maisi\or hunisi\or
  hulisi\or o'gostosi\or septemberi\or hoktemberi\or noyemberi\or
  dektemberi\fi~\number\year th.}
