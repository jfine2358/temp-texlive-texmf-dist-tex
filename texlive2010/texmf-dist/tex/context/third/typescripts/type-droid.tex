\starttypescriptcollection [droid]

\starttypescript [serif] [droid]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [DroidSerif-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [DroidSerif-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [DroidSerif-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [DroidSerif-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [droid]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [DroidSans-Regular] [features=default]
    \definefontsynonym [SansBold] [DroidSans-Bold]    [features=default]
\stoptypescript

\starttypescript [mono] [droid]
    \setups[font:fallback:mono]
    \definefontsynonym [Mono] [DroidMono-Regular] [features=default]
\stoptypescript

\starttypescript [droid]
    \definetypeface [droid] [rm] [serif] [droid]  [default]
    \definetypeface [droid] [ss] [sans]  [droid]  [default]
    \definetypeface [droid] [tt] [mono]  [droid]  [default]
    \definetypeface [droid] [mm] [math]  [modern] [default]
\stoptypescript

\stoptypescriptcollection
