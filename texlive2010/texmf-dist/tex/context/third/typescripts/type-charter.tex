\starttypescript [serif] [charter]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [CharterITC-Regular]    [features=default]
    \definefontsynonym [SerifBold]       [CharterITC-Bold]       [features=default]
    \definefontsynonym [SerifItalic]     [CharterITC-Italic]     [features=default]
    \definefontsynonym [SerifBoldItalic] [CharterITC-BoldItalic] [features=default]
\stoptypescript

\starttypescript [serif] [charter-bold]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [CharterITC-Bold]        [features=default]
    \definefontsynonym [SerifBold]       [CharterITC-Black]       [features=default]
    \definefontsynonym [SerifItalic]     [CharterITC-BoldItalic]  [features=default]
    \definefontsynonym [SerifBoldItalic] [CharterITC-BlackItalic] [features=default]
\stoptypescript

\starttypescript [charter,charter-bold]
    \definetypeface [\typescriptone] [rm] [serif] [\typescriptone] [default]
\stoptypescript

\loadmarkfile{type-charter}
