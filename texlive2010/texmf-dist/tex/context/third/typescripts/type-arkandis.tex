% Accanthis

\starttypescript [serif] [accanthis]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [Accanthis-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [Accanthis-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [Accanthis-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [Accanthis-BoldItalic] [features=default]
\stoptypescript

\starttypescript [accanthis]
    \definetypeface [accanthis] [rm] [serif] [accanthis] [default]
\stoptypescript
