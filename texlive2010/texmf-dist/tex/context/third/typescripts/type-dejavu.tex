\starttypescriptcollection [dejavu]

\starttypescript [serif] [dejavu]
    \setups[font:fallback:sans]
    \definefontsynonym [Serif]           [DejaVuSerif-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [DejaVuSerif-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [DejaVuSerif-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [DejaVuSerif-BoldItalic] [features=default]
\stoptypescript

\starttypescript [serif] [dejavu-cond]
    \setups[font:fallback:sans]
    \definefontsynonym [Serif]           [DejaVuSerifCond-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [DejaVuSerifCond-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [DejaVuSerifCond-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [DejaVuSerifCond-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [dejavu]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [DejaVuSans-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [DejaVuSans-Italic]     [features=default]
    \definefontsynonym [SansBold]       [DejaVuSans-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [DejaVuSans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [dejavu-cond]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [DejaVuSansCond-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [DejaVuSansCond-Italic]     [features=default]
    \definefontsynonym [SansBold]       [DejaVuSansCond-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [DejaVuSansCond-BoldItalic] [features=default]
\stoptypescript

\starttypescript [mono] [dejavu]
    \setups[font:fallback:mono]
    \definefontsynonym [Mono]           [DejaVuMono-Regular]    [features=default]
    \definefontsynonym [MonoItalic]     [DejaVuMono-Italic]     [features=default]
    \definefontsynonym [MonoBold]       [DejaVuMono-Bold]       [features=default]
    \definefontsynonym [MonoBoldItalic] [DejaVuMono-BoldItalic] [features=default]
\stoptypescript

\starttypescript [dejavu,dejavu-cond]
    \definetypeface [\typescriptone] [rm] [serif] [\typescriptone] [default]
    \definetypeface [\typescriptone] [ss] [sans]  [\typescriptone] [default]
    \definetypeface [\typescriptone] [tt] [mono]  [dejavu]         [default]
    \definetypeface [\typescriptone] [mm] [math]  [palatino]       [default]
\stoptypescript

\stoptypescriptcollection
