%D exljbris is a collection of fonts by Jos Buivenga. The fonts are
%D \startitemize
%D \head Anivers \par
%D       Anivers is a sans serif font that comes in regular, small caps,
%D       italic, and bold. Anivers Regular is free.
%D \head Delicious \par
%D       Delicious is a sans serif font that comes in regular, small caps,
%D       italic, and bold. All variants are free.
%D \head Diavlo \par
%D       Diavlo is a sans serif font that comes in light, book, medium,
%D       bold, and black. All variants are free. 
%D \head Fertigo Pro \par
%D       Fertigo Pro is a sans serif font that comes in regular weight. It
%D       is available freely.
%D \head Fontin and Fontin Sans \par
%D       Fontin is a serif font and Fontin Sans is a matching sans serif
%D       font. Both come in regular, small caps, italic, and bold. Both
%D       fonts can be downloaded freely.
%D \head Museo and Museo Sans \par
%D       Museo is a serif font and Museo Sans is a serif font. Both fonts comes
%D       in five weights (100 300 500 700 900). Museo Sans also has each
%D       weight in italic.
%D
%D       For Museo, three weights are free (300 500 700) and two are paid
%D       (100 900). For Museo Sans, regular and italic at weight 500 are
%D       free; other eight are paid.
%D \head Tallys \par
%D       Tallys is a serif font that comes in regular. It can be downloaded
%D       freely.
%D \stopitemize
%D
%D More details, including links for download, are available at
%D \hypenatedurl{http://www.josbuivenga.demon.nl}

\starttypescriptcollection [exljbris]

% Anivers

\starttypescript [sans] [anivers]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans] [Anivers-Regular] [features=default]
\stoptypescript

\starttypescript [anivers]
    \definetypeface [anivers] [ss] [sans] [anivers] [default]
\stoptypescript

% Delicious

\starttypescript [sans] [delicious]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [Delicious-Regular]
    \definefontsynonym [SansItalic]     [Delicious-Italic]
    \definefontsynonym [SansBold]       [Delicious-Bold]
    \definefontsynonym [SansBoldItalic] [Delicious-BoldItalic]
    \definefontsynonym [SansCaps]       [Delicious-Caps]
\stoptypescript

\starttypescript [delicious]
    \definetypeface [delicious] [ss] [sans] [delicious] [default]
\stoptypescript

% Diavlo

\starttypescript [sans] [diavlo]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [DiavloBook-Regular]   [features=default]
    \definefontsynonym [SansBold] [DiavloMedium-Regular] [features=default]  
\stoptypescript

\starttypescript [sans] [diavlo-light]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [DiavloLight-Regular] [features=default]
    \definefontsynonym [SansBold] [DiavloBook-Regular]  [features=default]  
\stoptypescript

\starttypescript [sans] [diavlo-medium]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [DiavloMedium-Regular] [features=default]
    \definefontsynonym [SansBold] [DiavloBlack-Regular]  [features=default]  
\stoptypescript

\starttypescript [sans] [diavlo-bold]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [DiavloBold-Regular]  [features=default]
    \definefontsynonym [SansBold] [DiavloBlack-Regular] [features=default]  
\stoptypescript

\starttypescript [diavlo,diavlo-light,diavlo-medium,diavlo-bold]
    \definetypeface [\typescriptone] [ss] [sans] [\typescriptone] [default]
\stoptypescript

% Fertigo Pro

\starttypescript [sans] [fertigo]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans] [FertigoPro-Regular] [features=default]
\stoptypescript

\starttypescript [fertigo]
    \definetypeface [fertigo] [ss] [sans] [fertigo] [default]
\stoptypescript

% Fontin, Fontin Sans

\starttypescript [serif] [fontin]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [Fontin-Regular]
    \definefontsynonym [SerifItalic]     [Fontin-Italic]
    \definefontsynonym [SerifBold]       [Fontin-Bold]
    \definefontsynonym [SerifBoldItalic] [SerifBold]
    \definefontsynonym [SerifCaps]       [Fontin-Caps]
\stoptypescript

\starttypescript [sans] [fontin]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [FontinSans-Regular]
    \definefontsynonym [SansItalic]     [FontinSans-Italic]
    \definefontsynonym [SansBold]       [FontinSans-Bold]
    \definefontsynonym [SansBoldItalic] [FontinSans-BoldItalic]
    \definefontsynonym [SansCaps]       [FontinSans-Caps]
\stoptypescript

\starttypescript [fontin]
    \definetypeface [fontin] [rm] [serif] [fontin]   [default]
    \definetypeface [fontin] [ss] [sans]  [fontin]   [default]
    \definetypeface [fontin] [tt] [mono]  [modern]   [default] [rscale=1.1]
    \definetypeface [fontin] [mm] [math]  [palatino] [default]
\stoptypescript

% Museo, Museo Sans

\starttypescript [serif] [museo]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]     [Museo-500] [features=default]
    \definefontsynonym [SerifBold] [Museo-700] [features=default]
\stoptypescript

\starttypescript [sans] [museo]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]       [MuseoSans-500]        [features=default]
    \definefontsynonym [SansItalic] [MuseoSans-500-Italic] [features=default]
\stoptypescript

\starttypescript [museo]
    \definetypeface [museo] [rm] [serif] [museo] [default]
    \definetypeface [museo] [ss] [sans]  [museo] [default]
\stoptypescript

% Tallys

\starttypescript [serif] [tallys]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif] [Tallys-Regular] [features=default]
\stoptypescript

\starttypescript [tallys]
    \definetypeface [tallys] [rm] [serif] [tallys] [default]
\stoptypescript

\stoptypescriptcollection
