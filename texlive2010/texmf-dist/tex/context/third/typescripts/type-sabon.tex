\starttypescript [serif] [sabon]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [Sabon-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [Sabon-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [Sabon-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [Sabon-BoldItalic] [features=default]
\stoptypescript

\starttypescript [serif] [sabon]
    \definefontsynonym [Sabon-Regular]    [name:SabonLTStd-Roman]      [features=default]
    \definefontsynonym [Sabon-Italic]     [name:SabonLTStd-Italic]     [features=default]
    \definefontsynonym [Sabon-Bold]       [name:SabonLTStd-Bold]       [features=default]
    \definefontsynonym [Sabon-BoldItalic] [name:SabonLTStd-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sabon]
    \definetypeface [sabon] [rm] [serif] [sabon]    [default]
    \definetypeface [sabon] [ss] [sans]  [modern]   [default]
    \definetypeface [sabon] [tt] [mono]  [modern]   [default] [rscale=1.075]
    \definetypeface [sabon] [mm] [math]  [palatino] [default]
\stoptypescript
