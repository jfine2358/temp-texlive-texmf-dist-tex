\definetypescriptprefix [n:minion]     [Minion]
\definetypescriptprefix [n:myriad]     [Myriad]
\definetypescriptprefix [n:warnockpro] [WarnockPro]

\starttypescriptcollection [adobe]

% Minion

\starttypescript [serif] [minion]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [\typescriptprefix{n:\typescripttwo}-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [\typescriptprefix{n:\typescripttwo}-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [\typescriptprefix{n:\typescripttwo}-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [\typescriptprefix{n:\typescripttwo}-BoldItalic] [features=default]
\stoptypescript

\starttypescript [minion]
    \definetypeface [minion] [rm] [serif] [minion] [default]
    \definetypeface [minion] [ss] [sans]  [myriad] [default]
    \definetypeface [minion] [tt] [mono]  [modern] [default]
    \definetypeface [minion] [mm] [math]  [modern] [default]
\stoptypescript

% Myriad

\starttypescript [sans] [myriad]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [\typescriptprefix{n:\typescripttwo}-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [\typescriptprefix{n:\typescripttwo}-Italic]     [features=default]
    \definefontsynonym [SansBold]       [\typescriptprefix{n:\typescripttwo}-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [\typescriptprefix{n:\typescripttwo}-BoldItalic] [features=default]
\stoptypescript

\starttypescript [myriad]
    \definetypeface [myriad] [rm] [serif] [minion] [default]
    \definetypeface [myriad] [ss] [sans]  [myriad] [default]
    \definetypeface [myriad] [tt] [mono]  [modern] [default]
    \definetypeface [myriad] [mm] [math]  [modern] [default]
\stoptypescript

% Warnock

\starttypescript [serif] [warnockpro]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [\typescriptprefix{n:\typescripttwo}-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [\typescriptprefix{n:\typescripttwo}-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [\typescriptprefix{n:\typescripttwo}-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [\typescriptprefix{n:\typescripttwo}-BoldItalic] [features=default]
\stoptypescript

\starttypescript [warnockpro]
    \definetypeface [warnockpro] [rm] [serif] [warnockpro] [default]
\stoptypescript

\stoptypescriptcollection
