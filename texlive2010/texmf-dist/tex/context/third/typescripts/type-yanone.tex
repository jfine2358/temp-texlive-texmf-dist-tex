\starttypescriptcollection [yanone]

% Al Abdali 8

\starttypescript [sans] [alabdali]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans] [Kaffeesatz-Bold] [features=arabic]
\stoptypescript

\starttypescript [alabdali]
    \definetypeface [alabdali] [ss] [sans] [alabdali] [default]
\stoptypescript

% Kaffeesatz

\starttypescript [sans] [kaffeesatz]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Kaffeesatz-Regular] [features=default]
    \definefontsynonym [SansBold] [Kaffeesatz-Bold]    [features=default]
\stoptypescript

\starttypescript [sans] [kaffeesatz-thin]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Kaffeesatz-Thin]  [features=default]
    \definefontsynonym [SansBold] [Kaffeesatz-Light] [features=default]
\stoptypescript

\starttypescript [sans] [kaffeesatz-light]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Kaffeesatz-Light]   [features=default]
    \definefontsynonym [SansBold] [Kaffeesatz-Regular] [features=default]
\stoptypescript

\starttypescript [kaffeesatz,kaffeesatz-thin,kaffeesatz-light]
    \definetypeface [\typescriptone] [ss] [sans] [\typescriptone] [default]
\stoptypescript

% Tagesschrift

\starttypescript [serif] [tagesschrift]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif] [Tagesschrift-Regular]   [features=default]
\stoptypescript

\starttypescript [tagesschrift]
    \definetypeface [tagesschrift] [rm] [serif] [tagesschrift] [default]
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-yanone}
