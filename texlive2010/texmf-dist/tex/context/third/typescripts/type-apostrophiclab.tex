% Amerika / Amerika Sans

\starttypescript [serif] [amerika]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif] [AmerikaSerif-Regular] [features=default]
\stoptypescript

\starttypescript [sans] [amerika]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans] [AmerikaSans-Regular] [features=default]
\stoptypescript

\starttypescript [amerika]
    \definetypeface [amerika] [rm] [serif] [amerika] [default]
    \definetypeface [amerika] [ss] [sans]  [amerika] [default]
\stoptypescript
