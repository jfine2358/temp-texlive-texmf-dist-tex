\starttypescript [sans] [goudysans]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [GoudySans-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [GoudySans-Italic]     [features=default]
    \definefontsynonym [SansBold]       [GoudySans-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [GoudySans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [goudysans]
    \definefontsynonym [GoudySans-Regular]    [name:GoudySans-Regular]    [features=default]
    \definefontsynonym [GoudySans-Italic]     [name:GoudySans-Italic]     [features=default]
    \definefontsynonym [GoudySans-Bold]       [name:GoudySans-Bold]       [features=default]
    \definefontsynonym [GoudySans-BoldItalic] [name:GoudySans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [goudysans]
    \definetypeface [goudysans] [ss] [sans] [goudysans] [default]
\stoptypescript
