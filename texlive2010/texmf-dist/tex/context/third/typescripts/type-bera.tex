\starttypescriptcollection [bera]

\starttypescript [serif] [bera]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]     [BeraSerif-Regular] [features=default]
    \definefontsynonym [SerifBold] [BeraSerif-Bold]    [features=default]
\stoptypescript

\starttypescript [sans] [bera]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [BeraSans-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [BeraSans-Italic]     [features=default]
    \definefontsynonym [SansBold]       [BeraSans-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [BeraSans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [mono] [bera]
    \setups[font:fallback:mono]
    \definefontsynonym [Mono]           [BeraSansMono-Regular]    [features=default]
    \definefontsynonym [MonoItalic]     [BeraSansMono-Italic]     [features=default]
    \definefontsynonym [MonoBold]       [BeraSansMono-Bold]       [features=default]
    \definefontsynonym [MonoBoldItalic] [BeraSansMono-BoldItalic] [features=default]
\stoptypescript

\starttypescript [bera]
    \definetypeface [bera] [rm] [serif] [bera]   [default]
    \definetypeface [bera] [ss] [sans]  [bera]   [default]
    \definetypeface [bera] [tt] [mono]  [bera]   [default]
    \definetypeface [bera] [mm] [math]  [modern] [default]
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-bera}
