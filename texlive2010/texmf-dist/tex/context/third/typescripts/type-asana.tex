\starttypescriptcollection[asana]

\starttypescript [math] [asana] [name]
    \definefontsynonym [MathRoman] [name:Asana-Math] [features=math\mathsizesuffix]
\stoptypescript

\starttypescript [asana]
    \definetypeface [asana] [rm] [serif] [palatino] [default]
    \definetypeface [asana] [ss] [sans]  [modern]   [default] [rscale=1.075]
    \definetypeface [asana] [tt] [mono]  [modern]   [default] [rscale=1.075]
    \definetypeface [asana] [mm] [math]  [asana]    [default]
    \quittypescriptscanning
\stoptypescript

\stoptypescriptcollection
