\starttypescriptcollection [verajja]

\starttypescript [serif] [verajja]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [VerajjaSerif-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [VerajjaSerif-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [VerajjaSerif-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [VerajjaSerif-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [verajja]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [VerajjaSans-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [VerajjaSans-Italic]     [features=default]
    \definefontsynonym [SansBold]       [VerajjaSans-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [VerajjaSans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [verajja]
    \definetypeface [verajja] [rm] [serif] [verajja] [default]
    \definetypeface [verajja] [ss] [sans]  [verajja] [default]
    \definetypeface [verajja] [tt] [mono]  [modern]  [default]
    \definetypeface [verajja] [mm] [math]  [modern]  [default]
\stoptypescript

\stoptypescriptcollection
