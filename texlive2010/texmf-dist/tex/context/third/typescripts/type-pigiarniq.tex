\starttypescriptcollection [pigiarniq]

\starttypescript [sans] [pigiarniq]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]       [Pigiarniq-Regular] [features=default]
    \definefontsynonym [SansItalic] [Pigiarniq-Italic]  [features=default]
    \definefontsynonym [SansBold]   [Pigiarniq-Bold]    [features=default]
\stoptypescript

\starttypescript [sans] [pigiarniq-thin]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [Pigiarniq-Light]   [features=default]
    \definefontsynonym [SansBold]       [Pigiarniq-Regular] [features=default]
    \definefontsynonym [SansBoldItalic] [Pigiarniq-Italic]  [features=default]
\stoptypescript

\starttypescript [sans] [pigiarniq-bold]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Pigiarniq-Bold]  [features=default]
    \definefontsynonym [SansBold] [Pigiarniq-Heavy] [features=default]
\stoptypescript

\starttypescript [pigiarniq,pigiarniq-thin,pigiarniq-bold]
    \definetypeface [\typescriptone] [ss] [sans] [\typescriptone] [default]
\stoptypescript

\stoptypescriptcollection
