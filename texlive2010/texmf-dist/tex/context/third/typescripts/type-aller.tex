\starttypescriptcollection [aller]

\starttypescript [sans] [aller]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [Aller-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [Aller-Italic]     [features=default]
    \definefontsynonym [SansBold]       [Aller-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [Aller-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [aller-light]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]       [Aller-LightRegular] [features=default]
    \definefontsynonym [SansItalic] [Aller-LightItalic]  [features=default]
\stoptypescript

\starttypescript [sans] [aller-display]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans] [Aller-Display] [features=default]
\stoptypescript

\starttypescript [aller,aller-light,aller-display]
    \definetypeface [\typescriptone] [ss] [sans] [\typescriptone] [default]
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-aller}

\endinput
