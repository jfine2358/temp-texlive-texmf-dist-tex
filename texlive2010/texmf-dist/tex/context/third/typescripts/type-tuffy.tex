\starttypescript [sans] [tuffy]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [Tuffy-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [Tuffy-Italic]     [features=default]
    \definefontsynonym [SansBold]       [Tuffy-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [Tuffy-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [tuffy]
    \definefontsynonym [Tuffy-Regular]    [name:Tuffy Regular]     [features=default]
    \definefontsynonym [Tuffy-Italic]     [name:Tuffy Italic]      [features=default]
    \definefontsynonym [Tuffy-Bold]       [name:Tuffy Bold]        [features=default]
    \definefontsynonym [Tuffy-BoldItalic] [name:Tuffy Bold Italic] [features=default]
\stoptypescript

\starttypescript [tuffy]
    \definetypeface [tuffy] [ss] [sans] [tuffy] [default]
\stoptypescript
