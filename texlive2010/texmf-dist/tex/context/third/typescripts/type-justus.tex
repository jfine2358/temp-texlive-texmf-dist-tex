\starttypescript [serif] [justus]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]       [Justus-Regular] [features=default]
    \definefontsynonym [SerifItalic] [Justus-Italic]  [features=default]
    \definefontsynonym [SerifBold]   [Justus-Bold]    [features=default]
    \definefontsynonym [SerifCaps]   [Justus-Caps]    [features=default]
\stoptypescript

\starttypescript [serif] [justus-os]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]       [Justus-Oldstyle]       [features=default]
    \definefontsynonym [SerifBold]   [Justus-Bold]           [features=default]
    \definefontsynonym [SerifItalic] [Justus-ItalicOldstyle] [features=default]
    \definefontsynonym [SerifCaps]   [Justus-Caps]           [features=default]
\stoptypescript

\starttypescript [serif] [justus,justus-os]
    \definefontsynonym [Justus-Regular]        [name:Justus Roman]           [features=default]
    \definefontsynonym [Justus-Italic]         [name:Justus Italic]          [features=default]
    \definefontsynonym [Justus-Bold]           [name:Justus Bold]            [features=default]
    \definefontsynonym [Justus-Oldstyle]       [name:Justus Oldstyle]        [features=default]
    \definefontsynonym [Justus-ItalicOldstyle] [name:Justus Italic Oldstyle] [features=default]
    \definefontsynonym [Justus-Caps]           [name:Justus Versalitas]      [features=default]
\stoptypescript

\starttypescript [justus,justus-os]
    \definetypeface [\typescriptone] [rm] [serif] [\typescriptone] [default]
    \definetypeface [\typescriptone] [ss] [sans]  [modern]         [default]
    \definetypeface [\typescriptone] [tt] [mono]  [modern]         [default]
    \definetypeface [\typescriptone] [mm] [math]  [modern]         [default]
\stoptypescript

\endinput
