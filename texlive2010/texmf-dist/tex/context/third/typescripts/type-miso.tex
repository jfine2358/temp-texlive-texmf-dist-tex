\starttypescript [sans] [miso-light]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Miso-Light]   [features=default]
    \definefontsynonym [SansBold] [Miso-Regular] [features=default]
\stoptypescript

\starttypescript [sans] [miso]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]     [Miso-Regular] [features=default]
    \definefontsynonym [SansBold] [Miso-Bold]    [features=default]
\stoptypescript

\starttypescript [sans] [miso-light,miso]
    \definefontsynonym [Miso-Light]   [name:Miso Light] [features=default]
    \definefontsynonym [Miso-Regular] [name:Miso]       [features=default]
    \definefontsynonym [Miso-Bold]    [name:Miso Bold]  [features=default]
\stoptypescript

\starttypescript [miso-light,miso]
    \definetypeface [\typescriptone] [ss] [sans] [\typescriptone] [default]
\stoptypescript

\endinput
