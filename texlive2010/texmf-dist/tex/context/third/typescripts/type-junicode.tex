\starttypescript [serif] [junicode]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [Junicode-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [Junicode-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [Junicode-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [Junicode-BoldItalic] [features=default]
\stoptypescript

\starttypescript [serif] [junicode]
    \definefontsynonym [Junicode-Regular]    [name:JunicodeRegular]    [features=default]
    \definefontsynonym [Junicode-Italic]     [name:JunicodeItalic]     [features=default]
    \definefontsynonym [Junicode-Bold]       [name:JunicodeBold]       [features=default]
    \definefontsynonym [Junicode-BoldItalic] [name:JunicodeBoldItalic] [features=default]
\stoptypescript

\starttypescript [junicode]
    \definetypeface [junicode] [rm] [serif] [junicode] [default]
\stoptypescript
