\starttypescriptcollection[luxi]

\starttypescript [serif] [luxi]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]           [LuxiSerif-Regular]    [features=default]
    \definefontsynonym [SerifItalic]     [LuxiSerif-Italic]     [features=default]
    \definefontsynonym [SerifBold]       [LuxiSerif-Bold]       [features=default]
    \definefontsynonym [SerifBoldItalic] [LuxiSerif-BoldItalic] [features=default]
\stoptypescript

\starttypescript [sans] [luxi]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [LuxiSans-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [LuxiSans-Italic]     [features=default]
    \definefontsynonym [SansBold]       [LuxiSans-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [LuxiSans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [mono] [luxi]
    \setups[font:fallback:mono]
    \definefontsynonym [Mono]           [LuxiMono-Regular]    [features=default]
    \definefontsynonym [MonoItalic]     [LuxiMono-Italic]     [features=default]
    \definefontsynonym [MonoBold]       [LuxiMono-Bold]       [features=default]
    \definefontsynonym [MonoBoldItalic] [LuxiMono-BoldItalic] [features=default]
\stoptypescript

\starttypescript [luxi]
    \definetypeface [luxi] [rm] [serif] [luxi]   [default]
    \definetypeface [luxi] [ss] [sans]  [luxi]   [default]
    \definetypeface [luxi] [tt] [mono]  [luxi]   [default]
    \definetypeface [luxi] [mm] [math]  [modern] [default]
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-luxi}
