\starttypescript [serif] [kontrapunkt]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]       [Kontrapunkt-Regular] [features=default]
    \definefontsynonym [SerifItalic] [Kontrapunkt-Italic]  [features=default]
    \definefontsynonym [SerifBold]   [Kontrapunkt-Bold]    [features=default]
\stoptypescript

\starttypescript [serif] [kontrapunkt]
    \definefontsynonym [Kontrapunkt-Regular] [name:Kontrapunkt-Light]       [features=default]
    \definefontsynonym [Kontrapunkt-Italic]  [name:Kontrapunkt-LightItalic] [features=default]
    \definefontsynonym [Kontrapunkt-Bold]    [name:Kontrapunkt-Bold]        [features=default]
\stoptypescript

\starttypescript [kontrapunkt]
    \definetypeface [kontrapunkt] [rm] [serif] [kontrapunkt] [default]
\stoptypescript

\endinput
