\starttypescriptcollection [vera]

\starttypescript [serif] [vera]
    \setups[font:fallback:serif]
    \definefontsynonym [Serif]     [VeraSerif-Regular] [features=default]
    \definefontsynonym [SerifBold] [VeraSerif-Bold]    [features=default]
\stoptypescript

\starttypescript [sans] [vera]
    \setups[font:fallback:sans]
    \definefontsynonym [Sans]           [VeraSans-Regular]    [features=default]
    \definefontsynonym [SansItalic]     [VeraSans-Italic]     [features=default]
    \definefontsynonym [SansBold]       [VeraSans-Bold]       [features=default]
    \definefontsynonym [SansBoldItalic] [VeraSans-BoldItalic] [features=default]
\stoptypescript

\starttypescript [mono] [vera]
    \setups[font:fallback:mono]
    \definefontsynonym [Mono]           [VeraMono-Regular]    [features=default]
    \definefontsynonym [MonoItalic]     [VeraMono-Italic]     [features=default]
    \definefontsynonym [MonoBold]       [VeraMono-Bold]       [features=default]
    \definefontsynonym [MonoBoldItalic] [VeraMono-BoldItalic] [features=default]
\stoptypescript

\starttypescript [vera]
    \definetypeface [vera] [rm] [serif] [vera]   [default]
    \definetypeface [vera] [ss] [sans]  [vera]   [default]
    \definetypeface [vera] [tt] [mono]  [vera]   [default]
    \definetypeface [vera] [mm] [math]  [modern] [default]
\stoptypescript

\stoptypescriptcollection

\loadmarkfile{type-vera}
