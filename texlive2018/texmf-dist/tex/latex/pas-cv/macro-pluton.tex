\def\modele@pluton{pluton}
\def\cv@h@pluton{5}

% =============== FOND ===============
\newcommand*\fond@pluton{%
	\fill[bottom color=\cmdPAS@bg@bgcolor!20,top color=white] (0,0) rectangle+(\paperwidth,-\cv@h@pluton);
	\draw[\cmdPAS@bg@bgcolor] (0,-\cv@h@pluton) -- (\paperwidth,-\cv@h@pluton);
	\draw[\cmdPAS@bg@bgcolor!50,very thick] (.8\paperwidth,-2) circle (5mm);
	\fill[\cmdPAS@bg@bgcolor!20] ($(.8\paperwidth,-2)+(8mm,8mm)$) circle (2mm);
	\fill[\cmdPAS@bg@bgcolor!40] (0,0) ellipse[x radius=9cm,y radius=13cm];%5 & 8
	\foreach \rx/\ry/\c in {8.7/12.7/10,8.66/12.66/30,8.62/12.62/50,8.58/12.58/70,8.54/12.54/90}
	{
		\fill[fill=black!\c!\cmdPAS@bg@bgcolor!40] (0,0) ellipse[x radius=\rx cm,y radius=\ry cm];
	}
	\fill[white] (0,0) ellipse[x radius=8.5cm,y radius=12.5cm];
	\fill[\cmdPAS@bg@bgcolor!20] (3,-25) circle (3cm);
	\fill[white] (3,-25) circle (2.5cm);
	\fill[\cmdPAS@bg@bgcolor!30] (2.2,-17) circle (.7cm);
	\fill[white] (2.2,-17) circle (0.6cm);
	\draw[very thick,\cmdPAS@bg@bgcolor!50] (1cm,0) -- (1cm,-\paperheight);
	\draw[very thick,\cmdPAS@bg@bgcolor!50] (1.2cm,0) -- (1.2cm,-\paperheight);
	\node[color=\cmdPAS@bg@postecolor,below right] at (9cm,-.5*\cv@h@pluton) {\begin{minipage}{12cm}\scshape\cmdPAS@bg@postesize\bfseries\cmdPAS@bg@poste\end{minipage}};
	\clip (0,0) ellipse[x radius=8.5cm,y radius=12.5cm];
	\fill[\cmdPAS@bg@bordercolor] (0,-1) rectangle+(15,-1);
	\ifx\cmdPAS@bg@pic\@empty%
	\else
		\ifx\cmdPAS@bg@borderpic\@empty
			\node[above left] at (3.5,-12) {\includegraphics[scale=\cmdPAS@bg@scalepic]{\cmdPAS@bg@pic}};
		\else
			\node[above left,draw=\cmdPAS@bg@borderpic,very thick,inner sep=1pt] at (3.5,-12) {\includegraphics[scale=\cmdPAS@bg@scalepic]{\cmdPAS@bg@pic}};
		\fi
	\fi
}

		}
