%=============== INFO GAUCHE ===============
\newcommand*\infoLeft@architecte[1]{%
	\AddToShipoutPicture
	{%
		\ifx\cmdPAS@bg@pic\@empty
			\put(\LenToUnit{0.5cm},\LenToUnit{.97\paperheight})
			{%
				\rlap{
					\begin{minipage}[t]{0.5\paperwidth}
						#1
					\end{minipage}
				}
			}
		\else
			\put(\LenToUnit{3.5cm},\LenToUnit{.97\paperheight})
			{%
				\rlap
				{
					\begin{minipage}[t]{0.5\paperwidth}
						#1
					\end{minipage}
				}
			}
		\fi
	}
}

	\end{tikzpicture}
}
