\def\unihangulchar@@@{%
  \getthish@ngulcl@ss
  \dobeforeh@ngulch@r
  \leavevmode %%% moved from inside group
  \begingroup
    \geth@ngulpl@ne@ndch@r
    \font\font@name\h@font@name\h@ngulpl@ne\space\h@font@size
    \font@name
    \dhucs@check@if@char@exists
    \dhucs@emph\h@ngulch@r
    \h@ngulch@r
  \endgroup
  \do@fterh@ngulch@r
}
