%                       Hyphenating Words
%                       ----------- -----


\def \*{%
  \discretionary{}{}{}}

\def \hyphen{%
  \nobreak -\zposthyph{2}}

\def \slash {/\zposthyph{1}}


% These macros are used in ZZHMODE to allow special characters to hyphenate.

\def \zprehyph #1{%                                     {\parameter}
  \ifcase #1%
    \relax
  \or
    \relax
  \else
    \nobreak
  \fi}

\def \zposthyph #1{%                                    {\parameter}
  \ifcase #1%
    \relax
  \or
    \discretionary{}{}{}%
  \else
    \penalty \exhyphenpenalty \hskip 0pt\relax
  \fi}

%                       Displaying Hyphens
%                       ---------- -------


\def \showhyphens #1{%
  \setbox\zboxa = \vbox{%
    \hsize = \maxdimen
    \parfillskip = 0pt
    \pretolerance = -1
    \tolerance = -1
    \hbadness = 0pt
    \showboxdepth = 0
    \ #1}}
