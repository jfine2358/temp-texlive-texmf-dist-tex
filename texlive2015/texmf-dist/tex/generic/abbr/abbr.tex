\def\<#1>{%
    \expandafter\ifx\csname<#1>\endcsname\relax
        \errmessage{abbreviation <#1> undefined!}%
    \else
        \csname<#1>\endcsname
    \fi
}

\def\abbrA#1#2#3{%
    \expandafter\def\csname<#1>\endcsname{#2}%
}

\def\abbrB#1#2#3{%
    \def\abbrdef{#3}%
    \ifx\abbrdef\empty
        \expandafter\def\csname<#1>\endcsname{#2}%
    \else
        \expandafter\def\csname<#1>\endcsname{#3}%
    \fi
}
