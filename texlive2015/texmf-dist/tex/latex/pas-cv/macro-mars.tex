	\filldraw[fill=white,draw=\cmdPAS@bg@bgcolor,very thick] (-1,-\cv@h@mars) rectangle+(23,-.25);
	\filldraw[fill=white,draw=\cmdPAS@bg@bgcolor,very thick] (-1,-\cv@h@mars-.25) rectangle+(23,-.35);
	\fill[\cmdPAS@bg@bgcolor] (0,-\cv@h@mars-.6) rectangle+(\paperwidth,-1.5);
	\filldraw[fill=white,draw=\cmdPAS@bg@bgcolor,very thick] (-1,-\cv@h@mars-2.1) rectangle+(23,-.25);
	\ifx\cmdPAS@bg@pic\@empty%
	\else
		\ifx\cmdPAS@bg@borderpic\@empty
			\node[below] at (2,-1) {\includegraphics[scale=\cmdPAS@bg@scalepic]{\cmdPAS@bg@pic}};
		\else
			\node[below,draw=\cmdPAS@bg@borderpic,very thick,inner sep=1pt] (photo) at (2,-1) {\includegraphics[scale=\cmdPAS@bg@scalepic]{\cmdPAS@bg@pic}};

	\fill[white] (poste.east) circle (.75cm);
	\fill[white] (poste.west) circle (.75cm);
	\fill[white] ($(poste.west)+(0,.8)$) rectangle ($(poste.east)+(0,-.8)$);
	\node[color=\cmdPAS@bg@postecolor,left] at (.95\paperwidth,-\cv@h@mars-1.35) {\scshape\cmdPAS@bg@postesize\bfseries\cmdPAS@bg@poste};
}

%=============== INFO GAUCHE ===============
\newcommand*\infoLeft@mars[1]{%
	\AddToShipoutPicture{%
		\put(\LenToUnit{4cm},\LenToUnit{.97\paperheight})
			{%
				\rlap{
					\begin{minipage}[t]{0.5\paperwidth}
						#1
					\end{minipage}
				}
			}
	}
}

%=============== INFO DROITE ===============
\newcommand*\infoRight@mars[1]{%
	\AddToShipoutPicture{%
		\put(\LenToUnit{.97\paperwidth},\LenToUnit{.97\paperheight}){%
			\llap{%
			\begin{minipage}[t]{.5\paperwidth}
				\begin{flushright}#1\end{flushright}
			\end{minipage}
			}
		}
	}
}

\newcommand*\title@mars[1]{%
	\noindent
	\begin{tikzpicture}
		\draw[\cmdPAS@title@bordercolor,very thick] (0,0) -- (\paperwidth,0);
		\node[above right,text=\cmdPAS@title@color] at (3,0) {\scshape\bfseries #1};
		\filldraw[draw=\cmdPAS@title@bordercolor,fill=white] (2,0) circle (.5cm);
	\end{tikzpicture}
}
